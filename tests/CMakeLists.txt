foreach(script ${TEST_SCRIPTS})
    get_filename_component(name ${script} NAME_WE)
    add_test(NAME ${name}
             COMMAND bash ${CMAKE_CURRENT_SOURCE_DIR}/${script})

        WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})
endforeach()
